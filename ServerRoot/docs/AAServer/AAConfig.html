def self.load_configuration
  return false if config_file_exists?
  begin
    tempConfig = YAML.load_file(@@options['configFile'])

    @@configs = tempConfig
    return true
  rescue => error
    puts "Configuration failed loading 'Config.yml' \n #{error}"
  end
  return false
end
